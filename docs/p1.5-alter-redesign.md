## 2. High-Value Turn Detector

### 設計思想

> 全部のターンで増やせとは言わない。ただし高価値ターンだけは追加知能コストを払う価値がある。

```typescript
export type TurnBudget = "standard" | "elevated" | "critical";

export interface TurnValueAssessment {
  budget: TurnBudget;
  reason: string;
  /** elevated/critical の場合、Insight Generator を起動するか */
  invoke_insight_generator: boolean;
}

/**
 * High-Value Turn Detector: このターンにどれだけの知能予算を割くか判定。
 * ルールベース。LLM呼び出しなし。
 */
export function assessTurnValue(
  signal: TurnSignal,
  arena: WinningInterpretation,
  model: LivingSelfModel,
  workingState: WorkingState | null,
  repairContext: RepairContext,
): TurnValueAssessment {

  // ── critical: 最高知能予算 ──

  // repair 直後（信頼回復の一打が最重要）
  if (repairContext.triggered) {
    return { budget: "critical", reason: "repair_triggered", invoke_insight_generator: true };
  }
  // 2ターン以上の連続co_think（深い共同思考の真っ最中）
  if (signal.intent === "co_think_request" && workingState && workingState.turn_count >= 2) {
    return { budget: "critical", reason: "deep_co_think", invoke_insight_generator: true };
  }
  // 「本質」「根本」「本当にしたいこと」「結局」— 存在的な問い
  if (/本質|根本|結局|本当[のにはが].*(?:したい|欲しい|大事|大切|求め)|人生|生き方|自分.*(?:何者|意味)/.test(signal.explicit.join(""))) {
    return { budget: "critical", reason: "existential_question", invoke_insight_generator: true };
  }

  // ── elevated: 追加知能あり ──

  // core_drive_check が Arena で勝った
  if (arena.primary.lens === "core_drive_check") {
    return { budget: "elevated", reason: "core_drive_arena_win", invoke_insight_generator: true };
  }
  // identity_quest が Arena で勝った
  if (arena.primary.lens === "identity_quest") {
    return { budget: "elevated", reason: "identity_quest", invoke_insight_generator: true };
  }
  // co_think_request の初回ターン
  if (signal.intent === "co_think_request" && (!workingState || workingState.turn_count <= 1)) {
    return { budget: "elevated", reason: "co_think_start", invoke_insight_generator: true };
  }
  // repeated return が直近で強く出ている
  if (model.repeated_returns.some(r => r.frequency >= 3 && computeRecency(r.last_seen) > 0.7)) {
    return { budget: "elevated", reason: "strong_repeated_return", invoke_insight_generator: true };
  }
  // protest 後の最初の通常ターン（信頼回復チャンス）
  if (workingState?.last_bet_outcome === "miss") {
    return { budget: "elevated", reason: "post_bet_miss", invoke_insight_generator: true };
  }

  // ── standard: 通常予算 ──
  return { budget: "standard", reason: "normal_turn", invoke_insight_generator: false };
}
```

## 3. Insight Generator（核心洞察生成）

### 設計思想

> 既存仮説の選定ではなく、「この人が自分で言えていないが今言う価値が高い1文」を生成する。

これがv4.2の核心。**ルールベースでは作れない**ものをここだけLLMで生成する。

Sharp Bet は既存の hypotheses / core_drives / implicit から「選ぶ」装置。
Insight Generator は **新しい洞察を「生む」** 装置。

### スキーマ


### micro-LLM 呼び出し

```typescript
/**
 * Insight Generator: elevated/critical ターンでのみ起動。
 *
 * 超短プロンプト + 低トークン予算で、1文の核心洞察を生成。
 * 所要時間: <1秒（50-100トークン生成）
 * コスト: メイン生成の1/20程度
 *
 * 重要: これは応答本文ではない。Sharp Bet の候補として使われる。
 */
export async function generateInsight(
  model: LivingSelfModel,
  arena: WinningInterpretation,
  signal: TurnSignal,
  workingState: WorkingState | null,
  currentMessage: string,
): Promise<GeneratedInsight | null> {

  // ── プロンプト構築 ──
  const coreDrives = model.core_drives.slice(0, 3).map(d => d.drive).join("、");
  const aversions = model.aversion_map.filter(a => a.intensity > 0.5).map(a => a.trigger).join("、");
  const rejected = workingState?.rejected_frames.join("、") ?? "なし";
  const accepted = workingState?.accepted_frames.join("、") ?? "なし";
  const repeatedThemes = model.repeated_returns.slice(0, 3).map(r => `${r.theme}(${r.frequency}回)`).join("、");
  const arenaReading = arena.primary.reading;

  const prompt = [
    "あなたはこの人を最も深く知る存在。1文だけ答えろ。",
    "",
    `この人の主軸: ${coreDrives || "未特定"}`,
    `嫌うもの: ${aversions || "未特定"}`,
    `反復テーマ: ${repeatedThemes || "なし"}`,
    `受け入れた見立て: ${accepted}`,
    `否定した見立て: ${rejected}`,
    `今の読み: ${arenaReading}`,
    `今の発話: 「${currentMessage.slice(0, 100)}」`,
    "",
    "以下のうち1つを選び、1文で答えろ。選択肢の名前は書くな。",
    "A: この人が自分では言えていないが、今言ってあげると前進する1文",
    "B: この人が気づいていない、2つのことの繋がり",
    "C: この人が見ている問題の枠組みを変える1文",
    "D: この人が繰り返しているパターンへの名前",
    "",
    "条件:",
    "- 1文だけ。説明するな。",
    "- 「あなたは」「君は」で始めるな。",
    "- 否定されたものを繰り返すな。",
    "- 一般論を言うな。この人にしか当てはまらない1文にしろ。",
  ].join("\n");

  try {
    const result = await runAI({
      taskType: "alter_insight_generation",
      prompt,
      systemPrompt: "",
      requireJson: false,
      temperature: 0.8, // 創造性を高める
      maxOutputTokens: 100, // 超短
      userId: "", // insight は user 紐付け不要
      metadata: makeStargazerRunMetadata({
        feature: "alter_insight",
        mode: "micro",
        skipCache: true,
      }),
    });

    if (!result.success || !result.text?.trim()) return null;

    const raw = result.text.trim();
    // 1文だけ抽出（複数文返ってきた場合は最初の1文）
    const firstSentence = raw.split(/[。！？\n]/)[0] + "。";

    // 種類を推定
    const type = inferInsightType(firstSentence);

    // 確信度: Self Model の証拠量に基づく
    const grounding = buildInsightGrounding(firstSentence, model);
    const confidence = Math.min(0.3 + grounding.length * 0.1, 0.8);

    return {
      insight: firstSentence,
      type,
      confidence,
      grounding,
      raw_output: raw,
    };
  } catch (e) {
    console.warn("[insight-generator] Failed:", e);
    return null; // graceful degradation: 失敗しても通常フローは続行
  }
}

function inferInsightType(insight: string): GeneratedInsight["type"] {
  if (/繋が|つなが|同じ|共通|どちらも/.test(insight)) return "hidden_connection";
  if (/実は|本当は|裏.*は|見えて.*いない/.test(insight)) return "unspoken_core";
  if (/ではなく|じゃなくて|問題は.*ではない|本題は/.test(insight)) return "reframe";
  if (/パターン|繰り返|毎回|いつも/.test(insight)) return "pattern_name";
  if (/矛盾|逆に|一方で.*他方/.test(insight)) return "contradiction_resolution";
  return "unspoken_core";
}

function buildInsightGrounding(insight: string, model: LivingSelfModel): string[] {
  const grounding: string[] = [];
  // Core Drives との関連
  for (const d of model.core_drives) {
    const keywords = d.drive.match(/[ぁ-ん]{3,}|[一-龥]{2,}|[a-zA-Z]{4,}/g) ?? [];
    if (keywords.some(kw => insight.includes(kw))) {
      grounding.push(`Core Drive: ${d.drive}`);
    }
  }
  // Repeated Returns との関連
  for (const r of model.repeated_returns) {
    if (insight.includes(r.theme.slice(0, 5))) {
      grounding.push(`Repeated Return: ${r.theme} (${r.frequency}回)`);
    }
  }
  // Meaning Patterns との関連
  for (const m of model.meaning_patterns) {
    if (insight.includes(m.concept)) {
      grounding.push(`Meaning: ${m.concept} = ${m.how_they_define_it}`);
    }
  }
  return grounding;
}
```

## 4. Claim Strength Controller

### 設計思想

> Truth Ownership が守りに効きすぎて Alter が弱くなる危険がある。
> いつ assert し、いつ probe するかの制御が必要。

```typescript
export type ClaimStrength = "assert" | "lean_in" | "probe" | "hold";

export interface ClaimDecision {
  strength: ClaimStrength;
  reason: string;
  /** assert の場合のフレーズガイド */
  phrase_guide: string;
}

/**
 * Claim Strength Controller: 仮説や洞察をどれだけ強く言い切るかを制御。
 *
 * 4段階:
 * - assert: 言い切る。「これは〜だと思う」「はっきり言うと〜」
 * - lean_in: 少し踏み込む。「僕の読みだと〜」「たぶん〜」
 * - probe: 問いかけ。「もしかして〜？」「〜じゃない？」
 * - hold: 今は言わない。温める。
 */
export function determineClaimStrength(
  bet: SharpBetWithFalsifier | null,
  model: LivingSelfModel,
  workingState: WorkingState | null,
  signal: TurnSignal,
): ClaimDecision {
  if (!bet) {
    return { strength: "hold", reason: "no_bet", phrase_guide: "" };
  }

  let strength: ClaimStrength = "probe"; // デフォルトは問いかけ

  // ── assert に昇格する条件 ──

  // 条件1: repeated return が3回以上 + confidence > 0.6
  const isRepeatedReturn = model.repeated_returns.some(r =>
    r.frequency >= 3 && bet.bet.includes(r.theme.slice(0, 5))
  );
  if (isRepeatedReturn && bet.confidence > 0.6) {
    strength = "assert";
  }

  // 条件2: Arena の conviction が高い（>0.4）+ confidence > 0.5
  // (conviction は arena から渡す必要があるが、bet.why_this_bet に含まれている)
  if (bet.confidence > 0.5 && bet.value_if_right === "transformative") {
    if (strength !== "assert") strength = "lean_in";
  }

  // 条件3: ユーザーが前ターンで building_on（同意して発展）
  if (signal.feedback_on_last_turn === "building_on" || signal.feedback_on_last_turn === "satisfied") {
    if (strength === "probe") strength = "lean_in";
    if (strength === "lean_in") strength = "assert";
  }

  // 条件4: accepted_frames に関連するもの（既に受け入れられた見立ての延長）
  if (workingState?.accepted_frames.some(f =>
    bet.bet.split(/\s/).some(w => w.length >= 3 && f.includes(w))
  )) {
    if (strength === "probe") strength = "lean_in";
  }

  // ── probe に降格する条件 ──

  // 条件A: trust_level < 2
  if (model.trust_level < 2 && strength === "assert") {
    strength = "lean_in";
  }
  if (model.trust_level < 1 && strength === "lean_in") {
    strength = "probe";
  }

  // 条件B: 前ターンで miss（外した直後は弱める）
  if (workingState?.last_bet_outcome === "miss") {
    if (strength === "assert") strength = "lean_in";
    if (strength === "lean_in") strength = "probe";
  }

  // 条件C: rejected_frames に近い内容（否定された見立ての延長は慎重に）
  if (workingState?.rejected_frames.some(f =>
    bet.bet.split(/\s/).some(w => w.length >= 3 && f.includes(w))
  )) {
    strength = "probe";
  }

  // ── hold: 今は言わない ──
  // risk_if_wrong が high + trust_level < 2
  if (bet.risk_if_wrong === "high" && model.trust_level < 2) {
    strength = "hold";
  }

  // フレーズガイド
  const phraseGuides: Record<ClaimStrength, string> = {
    assert: "「これは〜だと思う」「はっきり言うと」「間違いなく〜」— 確信を持って言い切る",
    lean_in: "「僕の読みだと〜」「たぶん〜」「見た感じ〜」— 少し踏み込むが断定はしない",
    probe: "「もしかして〜？」「〜じゃない？」「ひょっとして〜」— 問いかけとして出す",
    hold: "この仮説は今は出さない。温めて次のターンに持ち越す",
  };

  return {
    strength,
    reason: buildClaimReason(strength, bet, model),
    phrase_guide: phraseGuides[strength],
  };
}

function buildClaimReason(
  strength: ClaimStrength,
  bet: SharpBetWithFalsifier,
  model: LivingSelfModel,
): string {
  switch (strength) {
    case "assert":
      return `confidence=${bet.confidence.toFixed(2)} + repeated_returnまたはuser_agreement`;
    case "lean_in":
      return `confidence=${bet.confidence.toFixed(2)} + 高価値だが完全な確信ではない`;
    case "probe":
      return `confidence=${bet.confidence.toFixed(2)} + trust=${model.trust_level}で慎重に`;
    case "hold":
      return `risk_if_wrong=high + trust=${model.trust_level}で今は温める`;
  }
}
```

## 5. Elastic Intelligence Budget

### 全体像


- Insight micro call: 50-100トークン ≈ メイン生成の1/20のコスト

- 全体コスト増: **約5-15%**。レイテンシ増: **<1秒（micro callのみ）**

### Aha率の自動測定

```sql
-- Aha率: ユーザーの強い同意反応の割合
WITH user_turns AS (
  SELECT content FROM stargazer_alter_dialogues
  WHERE role = 'user' AND created_at >= NOW() - INTERVAL '7 days'
)
SELECT
  COUNT(*) FILTER (WHERE content ~ 'それだ|まさに|確かに|そうそう|当たってる|合ってる|その通り|わかる[！!]') * 100.0
    / NULLIF(COUNT(*), 0) AS aha_rate_pct,
  COUNT(*) AS total_user_turns
FROM user_turns;
```
